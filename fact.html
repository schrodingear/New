Program to Determine Factorial of a Number in C

#include <stdio.h>
#include <conio.h>

int main()
{
    int n, i, fact = 1;
    printf("Enter number: ");
    scanf("%d", &n); /* Reading n */

    if (n == 0 || n == 1)
        fact = 1;
    else if (n < 0)
        printf("Enter a positive integer");
    else
    {
        for (i = 1; i <= n; i++)
        {
            fact = fact * i;
        }
    }
    printf("The factorial of %d is %d", n, fact);
    getch();
}
